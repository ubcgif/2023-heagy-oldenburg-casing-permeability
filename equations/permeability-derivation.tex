\begin{displaymath}
    \begin{split}
    \nabla \times \vec{h} - \vec{j} &= \vec{j}_s \\
    \nabla \times \frac{1}{\mu} \vec{b} - \vec{j} &= \vec{j}_s \\
    \frac{1}{\mu} \nabla \times \vec{b} + \left(\nabla \frac{1}{\mu}\right) \times \vec{b} - \vec{j} &= \vec{j}_s
    \end{split}
\label{eq:permeability-derivation}
